## Goal

Deliver the full Web UI shell: Next.js 15 App Router with Fluent UI v9 `FluentProvider` (SSR-compatible Griffel), MSAL PKCE authentication (`@azure/msal-browser` v3 + `@azure/msal-react` v2), split-pane layout using `react-resizable-panels` (35/65 default, 25% min chat, 40% min dashboard), four right-pane tabs (Alerts default, Topology, Resources, Audit Log), multi-subscription `Combobox` selector in the top bar, desktop-only gate at 1200px, dark mode toggle, Dockerfile, and GitHub Actions CI workflow. Chat and dashboard panels are structural shells — populated in Plans 05-02 and 05-05.

**Handoff note:** `ChatPanel.tsx` created here is a minimal shell (empty state only). Plan 05-02 Task 5-02-05 owns the full ChatPanel implementation and will replace the shell entirely. `DashboardPanel.tsx` is similarly a shell replaced by Plan 05-05 Task 5-05-06. Do not add implementation logic to these shell files.

```typescript
'use client';

import React, { useState, useEffect, useMemo } from 'react';
import {
  FluentProvider,
  webLightTheme,
  webDarkTheme,
} from '@fluentui/react-components';
import { MsalProvider } from '@azure/msal-react';
import { getMsalInstance } from '@/lib/msal-instance';

type ThemeMode = 'light' | 'dark';

export function Providers({ children }: { children: React.ReactNode }) {
  const [themeMode, setThemeMode] = useState<ThemeMode>('light');
  const [msalReady, setMsalReady] = useState(false);

  const msalInstance = useMemo(() => getMsalInstance(), []);

  useEffect(() => {
    // Restore theme preference from localStorage
    const saved = localStorage.getItem('aap-theme') as ThemeMode | null;
    if (saved === 'dark' || saved === 'light') {
      setThemeMode(saved);
    }
    // Initialize MSAL
    msalInstance.initialize().then(() => {
      msalInstance.handleRedirectPromise().then(() => {
        setMsalReady(true);
      });
    });
  }, [msalInstance]);

  const theme = themeMode === 'dark' ? webDarkTheme : webLightTheme;

  if (!msalReady) {
    return (
      <FluentProvider theme={theme}>
        <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: '100vh' }}>
          Loading...
        </div>
      </FluentProvider>
    );
  }

  return (
    <MsalProvider instance={msalInstance}>
      <FluentProvider theme={theme}>
        {children}
      </FluentProvider>
    </MsalProvider>
  );
}
```

Note: The `setThemeMode` function is lifted to a context or prop in task 5-01-04 when the TopBar component needs to toggle dark mode. For this task, the state lives here and will be exposed via React context.

```typescript
'use client';

import React, { useState, useEffect } from 'react';
import { MessageBar, MessageBarBody, MessageBarTitle } from '@fluentui/react-components';

interface DesktopOnlyGateProps {
  minWidth: number;
  children: React.ReactNode;
}

export function DesktopOnlyGate({ minWidth, children }: DesktopOnlyGateProps) {
  const [isDesktop, setIsDesktop] = useState(true);

  useEffect(() => {
    const check = () => setIsDesktop(window.innerWidth >= minWidth);
    check();
    window.addEventListener('resize', check);
    return () => window.removeEventListener('resize', check);
  }, [minWidth]);

  if (!isDesktop) {
    return (
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: '100vh', padding: '2rem' }}>
        <MessageBar intent="warning">
          <MessageBarBody>
            <MessageBarTitle>Desktop Required</MessageBarTitle>
            This application requires a desktop browser (minimum 1200px width).
          </MessageBarBody>
        </MessageBar>
      </div>
    );
  }

  return <>{children}</>;
}
```

```typescript
import { useRouter } from 'next/navigation';

export default function AuthCallbackPage() {
  const { instance } = useMsal();

  useEffect(() => {
    instance.handleRedirectPromise().then((response) => {
      if (response) {
```
