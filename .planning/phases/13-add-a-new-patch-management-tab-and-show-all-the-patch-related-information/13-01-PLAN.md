```python
"""Patch Management endpoints — ARG queries for assessment and installations.

Exposes two async functions that query Azure Resource Graph for
PatchAssessmentResources and PatchInstallationResources tables.
Both handle skip_token pagination and return structured dicts
matching the response shapes from agents/patch/tools.py.
"""
from __future__ import annotations

import logging
from typing import Any, Dict, List, Optional

from azure.identity import DefaultAzureCredential
from azure.mgmt.resourcegraph import ResourceGraphClient
from azure.mgmt.resourcegraph.models import QueryRequest, QueryRequestOptions

logger = logging.getLogger(__name__)

# ---------------------------------------------------------------------------
# RBAC requirement: The API gateway managed identity (principal 69e05934-...)
# must have **Reader** role on every subscription passed to these queries.
# Without Reader, ARG queries return empty results or 403.
# Assign via: az role assignment create --assignee 69e05934-... --role Reader
#             --scope /subscriptions/<sub-id>
# ---------------------------------------------------------------------------

PATCH_ASSESSMENT_KQL = (
    "patchassessmentresources\n"
    '| where type =~ "microsoft.compute/virtualmachines/patchassessmentresults"\n'
    '    or type =~ "microsoft.hybridcompute/machines/patchassessmentresults"\n'
    "| extend rebootPending = tobool(properties.rebootPending),\n"
    "         osType = tostring(properties.osType),\n"
    "         lastAssessment = todatetime(properties.lastModifiedDateTime),\n"
    "         criticalCount = toint(properties.availablePatchCountByClassification.Critical),\n"
    "         securityCount = toint(properties.availablePatchCountByClassification.Security),\n"
    "         updateRollupCount = toint(properties.availablePatchCountByClassification.UpdateRollup),\n"
    "         featurePackCount = toint(properties.availablePatchCountByClassification.FeaturePack),\n"
    "         servicePackCount = toint(properties.availablePatchCountByClassification.ServicePack),\n"
    "         definitionCount = toint(properties.availablePatchCountByClassification.Definition),\n"
    "         toolsCount = toint(properties.availablePatchCountByClassification.Tools),\n"
    "         updatesCount = toint(properties.availablePatchCountByClassification.Updates)\n"
    "| project id, name, resourceGroup, subscriptionId, osType, rebootPending,\n"
    "          lastAssessment, criticalCount, securityCount, updateRollupCount,\n"
    "          featurePackCount, servicePackCount, definitionCount, toolsCount, updatesCount"
)

PATCH_INSTALLATIONS_KQL = (
    "patchinstallationresources\n"
    '| where type =~ "microsoft.compute/virtualmachines/patchinstallationresults"\n'
    '    or type =~ "microsoft.hybridcompute/machines/patchinstallationresults"\n'
    "| extend startTime = todatetime(properties.startDateTime),\n"
    "         status = tostring(properties.status),\n"
    "         rebootStatus = tostring(properties.rebootStatus),\n"
    "         installedCount = toint(properties.installedPatchCount),\n"
    "         failedCount = toint(properties.failedPatchCount),\n"
    "         pendingCount = toint(properties.pendingPatchCount),\n"
    "         startedBy = tostring(properties.startedBy)\n"
    "| where startTime > ago(7d)\n"
    "| project id, resourceGroup, subscriptionId, startTime, status,\n"
    "          rebootStatus, installedCount, failedCount, pendingCount, startedBy"
)

# Maximum rows per ARG request. Caps the pagination loop to prevent runaway
# queries on very large estates. ARG supports up to 1000 per page; we use 500
# as a safe default. Pagination via skip_token still fetches all results across
# multiple pages — this cap is per-page, not total.
_ARG_PAGE_SIZE = 500


def _run_arg_query(
    credential: DefaultAzureCredential,
    subscription_ids: List[str],
    kql: str,
) -> List[Dict[str, Any]]:
    """Execute an ARG query with skip_token pagination.

    Args:
        credential: Azure credential instance.
        subscription_ids: List of subscription IDs to scope the query.
        kql: KQL query string.

    Returns:
        Flat list of result dicts from all pages.
    """
    client = ResourceGraphClient(credential)
    all_results: List[Dict[str, Any]] = []
    skip_token: Optional[str] = None

    while True:
        options = QueryRequestOptions(
            skip_token=skip_token,
            top=_ARG_PAGE_SIZE,
        ) if skip_token else QueryRequestOptions(top=_ARG_PAGE_SIZE)
        request = QueryRequest(
            subscriptions=subscription_ids,
            query=kql,
            options=options,
        )
        response = client.resources(request)
        all_results.extend(response.data)

        skip_token = response.skip_token
        if not skip_token:
            break

    return all_results


async def query_patch_assessment(
    credential: DefaultAzureCredential,
    subscription_ids: List[str],
) -> Dict[str, Any]:
    """Query ARG PatchAssessmentResources for per-machine compliance data.

    Args:
        credential: Azure credential (from app.state via Depends).
        subscription_ids: Subscription IDs to query.

    Returns:
        Dict with keys: machines (list), total_count (int), query_status (str).
    """
    logger.info(
        "Querying patch assessment for %d subscription(s)", len(subscription_ids)
    )
    try:
        machines = _run_arg_query(credential, subscription_ids, PATCH_ASSESSMENT_KQL)
        return {
            "machines": machines,
            "total_count": len(machines),
            "query_status": "success",
        }
    except Exception as exc:
        logger.error("Patch assessment query failed: %s", exc)
        return {
            "machines": [],
            "total_count": 0,
            "query_status": "error",
            "error": str(exc),
        }


async def query_patch_installations(
    credential: DefaultAzureCredential,
    subscription_ids: List[str],
) -> Dict[str, Any]:
    """Query ARG PatchInstallationResources for 7-day installation history.

    Args:
        credential: Azure credential (from app.state via Depends).
        subscription_ids: Subscription IDs to query.

    Returns:
        Dict with keys: installations (list), total_count (int), days (int), query_status (str).
    """
    logger.info(
        "Querying patch installations for %d subscription(s)", len(subscription_ids)
    )
    try:
        installations = _run_arg_query(
            credential, subscription_ids, PATCH_INSTALLATIONS_KQL
        )
        return {
            "installations": installations,
            "total_count": len(installations),
            "days": 7,
            "query_status": "success",
        }
    except Exception as exc:
        logger.error("Patch installations query failed: %s", exc)
        return {
            "installations": [],
            "total_count": 0,
            "days": 7,
            "query_status": "error",
            "error": str(exc),
        }
```

Key design points:
- `PATCH_ASSESSMENT_KQL` and `PATCH_INSTALLATIONS_KQL` are module-level constants copied verbatim from `agents/patch/tools.py` lines 141-166 and 246-267 respectively.
- `_run_arg_query()` is a shared pagination helper (DRY — used by both endpoints).
- **RBAC comment** at module top documents the Reader role requirement for the gateway MI on target subscriptions.
- **Pagination cap:** `_ARG_PAGE_SIZE = 500` passed via `QueryRequestOptions(top=500)` to prevent unbounded page sizes on large estates. The skip_token loop still iterates all pages.
- Both async functions accept `credential` and `subscription_ids` as parameters (injected by route handlers in main.py).
- Error handling returns structured error responses (never raises), matching the pattern from the patch agent.
- Functions are `async` for consistency with FastAPI but the ARG SDK call is sync (acceptable for this use case).

```python
@app.get("/api/v1/patch/assessment")
async def get_patch_assessment(
    subscriptions: str,
    token: dict[str, Any] = Depends(verify_token),
    credential: DefaultAzureCredential = Depends(get_credential),
):
    """Query patch assessment data from Azure Resource Graph (D-01).

    Args:
        subscriptions: Comma-separated subscription IDs.

    Authentication: Entra ID Bearer token required.
    """
    subscription_ids = [s.strip() for s in subscriptions.split(",") if s.strip()]
    if not subscription_ids:
        raise HTTPException(status_code=400, detail="subscriptions parameter is required")
    result = await query_patch_assessment(credential, subscription_ids)
    if result.get("query_status") == "error":
        return JSONResponse(
            {"error": result.get("error", "ARG query failed"), "machines": [], "total_count": 0, "query_status": "error"},
            status_code=502,
        )
    return result


@app.get("/api/v1/patch/installations")
async def get_patch_installations(
    subscriptions: str,
    token: dict[str, Any] = Depends(verify_token),
    credential: DefaultAzureCredential = Depends(get_credential),
):
    """Query patch installation history from Azure Resource Graph (D-01).

    Args:
        subscriptions: Comma-separated subscription IDs.

    Authentication: Entra ID Bearer token required.
    """
    subscription_ids = [s.strip() for s in subscriptions.split(",") if s.strip()]
    if not subscription_ids:
        raise HTTPException(status_code=400, detail="subscriptions parameter is required")
    result = await query_patch_installations(credential, subscription_ids)
    if result.get("query_status") == "error":
        return JSONResponse(
            {"error": result.get("error", "ARG query failed"), "installations": [], "total_count": 0, "days": 7, "query_status": "error"},
            status_code=502,
        )
    return result
```

Both routes:
- Accept `subscriptions` as a required comma-separated query parameter (per D-02).
- Use `Depends(verify_token)` for Entra auth (same as all other endpoints).
- Use `Depends(get_credential)` to inject the shared `DefaultAzureCredential` (same as no other endpoint currently but matches the dependency pattern).
- Return HTTP 502 when ARG query fails (matching the proxy error pattern).
- Return HTTP 400 when `subscriptions` is empty.
