   ```python
   @app.get("/api/v1/patch/assessment")
   async def patch_assessment(
       subscriptions: str,
       token: dict[str, Any] = Depends(verify_token),
       credential: DefaultAzureCredential = Depends(get_credential),
   ):
       """Query ARG for patch assessment data across subscriptions (Phase 13).

       Args:
           subscriptions: Comma-separated subscription IDs to query.

       Authentication: Entra ID Bearer token required.
       """
       subscription_ids = [s.strip() for s in subscriptions.split(",") if s.strip()]
       if not subscription_ids:
           raise HTTPException(status_code=400, detail="subscriptions parameter is required")
       result = await query_patch_assessment(subscription_ids, credential)
       if result.get("query_status") == "error":
           error_msg = result.get("error", "Unknown error")
           if "permissions" in error_msg.lower() or "403" in error_msg:
               raise HTTPException(status_code=403, detail=error_msg)
           raise HTTPException(status_code=502, detail=error_msg)
       return result


   @app.get("/api/v1/patch/installations")
   async def patch_installations(
       subscriptions: str,
       token: dict[str, Any] = Depends(verify_token),
       credential: DefaultAzureCredential = Depends(get_credential),
   ):
       """Query ARG for patch installation history across subscriptions (Phase 13).

       Args:
           subscriptions: Comma-separated subscription IDs to query.

       Authentication: Entra ID Bearer token required.
       """
       subscription_ids = [s.strip() for s in subscriptions.split(",") if s.strip()]
       if not subscription_ids:
           raise HTTPException(status_code=400, detail="subscriptions parameter is required")
       result = await query_patch_installations(subscription_ids, credential)
       if result.get("query_status") == "error":
           error_msg = result.get("error", "Unknown error")
           if "permissions" in error_msg.lower() or "403" in error_msg:
               raise HTTPException(status_code=403, detail=error_msg)
           raise HTTPException(status_code=502, detail=error_msg)
       return result
   ```

Key points:
- Both use `Depends(verify_token)` for auth (matching all existing endpoints)
- Both use `Depends(get_credential)` to get the shared `DefaultAzureCredential` from `app.state`
- `subscriptions` is a required query parameter (comma-separated string), split into a list
- Empty subscriptions returns 400
- ARG permission errors return 403
- Other ARG errors return 502
