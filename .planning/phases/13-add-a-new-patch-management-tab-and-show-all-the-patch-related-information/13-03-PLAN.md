/**
 * Format an ISO date string as a relative time description (e.g., "2h ago", "3d ago").
 * No external dependency — pure arithmetic on Date objects.
 */
export function formatRelativeTime(isoString: string | null | undefined): string {
  if (!isoString) return '—';

  const date = new Date(isoString);
  if (isNaN(date.getTime())) return '—';

  const now = Date.now();
  const diffMs = now - date.getTime();

  if (diffMs < 0) return 'just now';

  const seconds = Math.floor(diffMs / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  const days = Math.floor(hours / 24);

  if (days > 0) return `${days}d ago`;
  if (hours > 0) return `${hours}h ago`;
  if (minutes > 0) return `${minutes}m ago`;
  return 'just now';
}
```

// Helpers

type ComplianceState = 'Compliant' | 'NonCompliant' | 'Unknown';

function deriveCompliance(m: AssessmentMachine): ComplianceState {
  if ((m.criticalCount ?? 0) > 0 || (m.securityCount ?? 0) > 0) return 'NonCompliant';
  if ((m.criticalCount ?? 0) === 0 && (m.securityCount ?? 0) === 0 && !m.rebootPending) return 'Compliant';
  return 'Unknown';
}

function extractMachineName(resourceId: string): string {
  // ARM ID: .../virtualMachines/<name>/patchInstallationResults/...
  const segments = resourceId.split('/');
  const idx = segments.findIndex(
    (s) => s.toLowerCase() === 'virtualmachines' || s.toLowerCase() === 'machines'
  );
  if (idx !== -1 && idx + 1 < segments.length) return segments[idx + 1];
  return segments[segments.length - 1] ?? resourceId;
}

function complianceHealth(pct: number): HealthStatus {
  if (pct >= 90) return 'healthy';
  if (pct >= 70) return 'warning';
  return 'critical';
}

function countHealth(count: number, warnThreshold: number, critThreshold: number): HealthStatus {
  if (count === 0) return 'healthy';
  if (count <= warnThreshold) return 'warning';
  return 'critical';
}

// ---------------------------------------------------------------------------
// Badge components
// ---------------------------------------------------------------------------

function ComplianceBadge({ state }: { state: ComplianceState }) {
  switch (state) {
    case 'Compliant':
      return <Badge className="bg-green-100 text-green-700 border-green-200 hover:bg-green-100">{state}</Badge>;
    case 'NonCompliant':
      return <Badge className="bg-red-100 text-red-700 border-red-200 hover:bg-red-100">{state}</Badge>;
    default:
      return <Badge variant="outline">{state}</Badge>;
  }
}

function StatusBadge({ status }: { status: string }) {
  switch (status) {
    case 'Succeeded':
      return <Badge className="bg-green-100 text-green-700 border-green-200 hover:bg-green-100">{status}</Badge>;
    case 'Failed':
      return <Badge variant="destructive">{status}</Badge>;
    case 'InProgress':
      return <Badge className="bg-blue-100 text-blue-700 border-blue-200 hover:bg-blue-100">{status}</Badge>;
    case 'CompletedWithWarnings':
      return <Badge className="bg-yellow-100 text-yellow-700 border-yellow-200 hover:bg-yellow-100">{status}</Badge>;
    default:
      return <Badge variant="outline">{status || 'Unknown'}</Badge>;
  }
}

      setInstallationsData(null);
      setLoading(false);
      return;

    setAssessmentError(null);
    setInstallationsError(null);

      fetch(`/api/proxy/patch/assessment?subscriptions=${encodeURIComponent(subsParam)}`, {
        signal: AbortSignal.timeout(15000),
      }).then(async (res) => {
        const data = await res.json();
        if (!res.ok) throw new Error(data?.error ?? `HTTP ${res.status}`);
        return data;
      }),
      fetch(`/api/proxy/patch/installations?subscriptions=${encodeURIComponent(subsParam)}`, {
        signal: AbortSignal.timeout(15000),
      }).then(async (res) => {
        const data = await res.json();
        if (!res.ok) throw new Error(data?.error ?? `HTTP ${res.status}`);
        return data;
      }),
    ]);

    if (assessmentResult.status === 'fulfilled') {
      setAssessmentData(assessmentResult.value.machines ?? []);
    } else {
      setAssessmentError(assessmentResult.reason?.message ?? 'Failed to load assessment data');
      setAssessmentData(null);

    if (installationsResult.status === 'fulfilled') {
      setInstallationsData(installationsResult.value.installations ?? []);
    } else {
      setInstallationsError(installationsResult.reason?.message ?? 'Failed to load installations data');
      setInstallationsData(null);
    }

    setLoading(false);
    hasFetchedRef.current = true;

  // Lazy load: only fetch when the tab is active (D-11).
  // First activation triggers initial fetch.

  useEffect(() => {
    if (!isActive) return;
    // Always re-fetch when subscriptions change while active

    fetchData();

  useEffect(() => {
    hasFetchedRef.current = false;

            </span>
          )}
        </div>
        <Button
          variant="outline"
          size="sm"
          onClick={() => fetchData()}
          disabled={loading}

          <RefreshCw className={`h-4 w-4 mr-1.5 ${loading ? 'animate-spin' : ''}`} />
          Refresh

                    <TableHead className="h-10 px-3 text-right text-[13px] font-semibold text-muted-foreground w-[80px]">Installed</TableHead>

                </TableHeader>
                <TableBody>
                  {installationsData.map((inst) => (
                    <TableRow key={`${inst.id}-${inst.startTime}`} className="border-b hover:bg-muted/30 transition-colors">

                      <TableCell className="h-10 px-3 align-middle text-[13px] text-muted-foreground">{formatRelativeTime(inst.startTime)}</TableCell>

                      <TableCell className="h-10 px-3 align-middle text-right font-mono text-[13px]">{inst.installedCount ?? 0}</TableCell>
                      <TableCell className={`h-10 px-3 align-middle text-right font-mono text-[13px] ${(inst.failedCount ?? 0) > 0 ? 'text-red-600' : ''}`}>{inst.failedCount ?? 0}</TableCell>
                      <TableCell className="h-10 px-3 align-middle text-right font-mono text-[13px]">{inst.pendingCount ?? 0}</TableCell>
                      <TableCell className="h-10 px-3 align-middle"><Badge variant="outline" className="text-xs">{inst.rebootStatus || '—'}</Badge></TableCell>
                      <TableCell className="h-10 px-3 align-middle text-[13px] text-muted-foreground truncate max-w-[150px]">{inst.startedBy || '—'}</TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
            </div>
          )}
        </div>
      )}

      {/* Loading skeleton for tables */}
      {loading && (
        <>
          <div className="flex flex-col gap-3">
            <Skeleton className="h-5 w-32" />
            <div className="flex flex-col gap-1">
              {Array.from({ length: 6 }).map((_, i) => (
                <Skeleton key={`a-${i}`} className="h-10 w-full" />
              ))}
            </div>
          </div>
          <div className="flex flex-col gap-3">
            <Skeleton className="h-5 w-48" />
            <div className="flex flex-col gap-1">
              {Array.from({ length: 6 }).map((_, i) => (
                <Skeleton key={`i-${i}`} className="h-10 w-full" />
              ))}
            </div>
          </div>
        </>
      )}
    </div>
  );
}
```

Key implementation details:
- **Props:** `subscriptions: string[]` and `isActive: boolean` — the `isActive` prop gates data fetching so the tab only loads when visible (D-11 lazy load).
- **Lazy load:** The `useEffect` is gated on `isActive` — data fetches ONLY when the tab is active. First activation triggers initial fetch. Subscription changes trigger re-fetch only when tab is active.
- **Parallel fetching:** `Promise.allSettled()` for independent error handling (D-12 partial data)
- **Separate error states:** `assessmentError` and `installationsError` (not one shared `error`)
- **Error copy:** Uses the UI-SPEC copywriting contract message: "Unable to load patch data. Check that the API gateway is running and the selected subscriptions have Azure Update Manager enabled." with a `<details>` block for the raw error.
- **5 MetricCards:** Total Machines, Compliant %, Critical+Security, Reboot Pending, Failed Installs — each with computed `HealthStatus`
- **Compliance derivation:** criticalCount + securityCount > 0 = NonCompliant; 0 + no reboot = Compliant; else Unknown
- **Machine name extraction:** `extractMachineName()` parses ARM resource ID
- **Client-side filters:** compliance Select + machine search Input (D-10)
- **No polling:** manual Refresh button only (D-11)
- **Empty/loading/error states:** all per UI-SPEC copywriting contract
