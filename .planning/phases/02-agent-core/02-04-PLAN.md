# Plan 02-04: Agent Implementations

> **Wave 4** — Depends on 02-02 (shared base image and shared utilities must exist).

## must_haves

1. Orchestrator agent uses `HandoffOrchestrator` with 6 `AgentTarget` registrations (compute, network, storage, security, sre, arc); classifies incidents by `domain` field and falls back to LLM classification when domain is ambiguous.
2. Each of 5 non-Arc domain agents (compute, network, storage, security, sre) implements the triage pattern: (1) query Activity Log for prior 2h changes, (2) query Log Analytics, (3) query Resource Health, (4) return typed `IncidentMessage` with `message_type: "diagnosis_complete"` containing `hypothesis`, `evidence`, and `confidence_score`.
3. Each domain agent decorates tools with `@ai_function` and registers explicit Azure MCP Server `allowed_tools` lists — no wildcard tool access.
4. Arc agent is a stub: responds with `{"status": "pending_phase3", "message": "Arc-specific capabilities are pending Phase 3 implementation..."}`.
5. All agents use `agents.shared.otel.setup_telemetry()` and `agents.shared.auth.get_credential()` — no direct `DefaultAzureCredential()` calls or hardcoded connection strings in agent code.
6. Each agent has a `Dockerfile` extending the shared base image via `ARG BASE_IMAGE` and a `requirements.txt` for agent-specific extras.
7. `.github/workflows/agent-images.yml` matrix-builds all 7 agent images on changes to `agents/{name}/`.

@ai_function
def classify_incident_domain(
    affected_resources: list[str],
    detection_rule: str,
    kql_evidence: str | None = None,
) -> str:
    """Classify an incident into a domain when the domain field is absent or ambiguous.

    Analyzes affected resource types and detection rule names to determine
    the correct domain agent. Returns one of: compute, network, storage,
    security, arc, sre.

    Args:
        affected_resources: List of ARM resource IDs.
        detection_rule: Name of the detection rule that fired.
        kql_evidence: Optional KQL query results.

    Returns:
        Domain string: compute, network, storage, security, arc, sre.
    """
    # Resource type to domain mapping for deterministic classification
    resource_domain_map = {
        "Microsoft.Compute": "compute",
        "Microsoft.ContainerService": "compute",
        "Microsoft.Web": "compute",
        "Microsoft.Network": "network",
        "Microsoft.Cdn": "network",
        "Microsoft.Storage": "storage",
        "Microsoft.DataLakeStore": "storage",
        "Microsoft.Security": "security",
        "Microsoft.KeyVault": "security",
        "Microsoft.HybridCompute": "arc",
        "Microsoft.Kubernetes": "arc",
    }

    for resource_id in affected_resources:
        for provider_prefix, domain in resource_domain_map.items():
            if provider_prefix.lower() in resource_id.lower():
                return domain

    # Fallback: SRE handles unclassified incidents
    return "sre"

"""Shared triage workflow base for all domain agents (TRIAGE-002, TRIAGE-003, TRIAGE-004).

Every domain agent MUST:
1. Query Activity Log for changes in prior 2 hours (TRIAGE-003)
2. Query Log Analytics for relevant events (TRIAGE-002)
3. Query Resource Health to distinguish platform vs config cause (TRIAGE-002)
4. Return hypothesis + evidence + confidence_score (TRIAGE-004)
"""
from __future__ import annotations

import json
import logging
from datetime import datetime, timezone
from typing import Any

from agents.shared.envelope import IncidentMessage

logger = logging.getLogger(__name__)


class TriageDiagnosis:
    """Structured triage diagnosis output (TRIAGE-004).

    Every domain agent must produce a TriageDiagnosis containing
    hypothesis, evidence, and confidence_score.
    """

    def __init__(
        self,
        hypothesis: str,
        evidence: list[dict[str, Any]],
        confidence_score: float,
        domain: str,
        affected_resources: list[str],
        activity_log_findings: list[dict[str, Any]] | None = None,
        resource_health_status: str | None = None,
        needs_cross_domain: bool = False,
        suspected_domain: str | None = None,
        remediation_proposal: dict[str, Any] | None = None,
    ):
        if not 0.0 <= confidence_score <= 1.0:
            raise ValueError(f"confidence_score must be 0.0-1.0, got {confidence_score}")

        self.hypothesis = hypothesis
        self.evidence = evidence
        self.confidence_score = confidence_score
        self.domain = domain
        self.affected_resources = affected_resources
        self.activity_log_findings = activity_log_findings or []
        self.resource_health_status = resource_health_status
        self.needs_cross_domain = needs_cross_domain
        self.suspected_domain = suspected_domain
        self.remediation_proposal = remediation_proposal

    def to_envelope(
        self,
        correlation_id: str,
        thread_id: str,
        source_agent: str,
        target_agent: str = "orchestrator",
    ) -> IncidentMessage:
        """Convert diagnosis to a typed IncidentMessage envelope (AGENT-002)."""
        now = datetime.now(timezone.utc).isoformat()
        return IncidentMessage(
            correlation_id=correlation_id,
            thread_id=thread_id,
            source_agent=source_agent,
            target_agent=target_agent,
            message_type="diagnosis_complete",
            payload={
                "hypothesis": self.hypothesis,
                "evidence": self.evidence,
                "confidence_score": self.confidence_score,
                "domain": self.domain,
                "affected_resources": self.affected_resources,
                "activity_log_findings": self.activity_log_findings,
                "resource_health_status": self.resource_health_status,
                "needs_cross_domain": self.needs_cross_domain,
                "suspected_domain": self.suspected_domain,
                "remediation_proposal": self.remediation_proposal,
            },
            timestamp=now,
        )

    def to_dict(self) -> dict[str, Any]:
        """Convert diagnosis to a plain dict."""
        return {
            "hypothesis": self.hypothesis,
            "evidence": self.evidence,
            "confidence_score": self.confidence_score,
            "domain": self.domain,
            "affected_resources": self.affected_resources,
            "activity_log_findings": self.activity_log_findings,
            "resource_health_status": self.resource_health_status,
            "needs_cross_domain": self.needs_cross_domain,
            "suspected_domain": self.suspected_domain,
            "remediation_proposal": self.remediation_proposal,
        }


class RemediationProposal:
    """Structured remediation proposal (REMEDI-001).

    No remediation is executed without explicit human approval.
    Every proposal includes description, target resources, estimated impact,
    risk level, and reversibility statement.
    """

    def __init__(
        self,
        description: str,
        target_resources: list[str],
        estimated_impact: str,
        risk_level: str,
        reversibility: str,
        action_type: str,
    ):
        valid_risk_levels = {"low", "medium", "high", "critical"}
        if risk_level not in valid_risk_levels:
            raise ValueError(f"risk_level must be one of {valid_risk_levels}, got '{risk_level}'")

        self.description = description
        self.target_resources = target_resources
        self.estimated_impact = estimated_impact
        self.risk_level = risk_level
        self.reversibility = reversibility
        self.action_type = action_type

    def to_dict(self) -> dict[str, Any]:
        return {
            "description": self.description,
            "target_resources": self.target_resources,
            "estimated_impact": self.estimated_impact,
            "risk_level": self.risk_level,
            "reversibility": self.reversibility,
            "action_type": self.action_type,
            "requires_approval": True,  # REMEDI-001: always
        }

    resource_ids: list[str],
