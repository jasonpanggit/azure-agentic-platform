    **Subscription resolution helper** (add as module-level function in each file, or import from a shared utility):

    ```python
    def _resolve_subscription_ids(
        subscriptions_param: Optional[str],
        request: Request,
    ) -> List[str]:
        """Resolve effective subscription IDs for ARG queries.

        Priority:
        1. Explicit subscriptions= query param (caller-specified subset)
        2. app.state.subscription_registry.get_all_ids() (all discovered)
        3. [] (empty — triggers graceful no-op in ARG queries)
        """
        if subscriptions_param:
            return [s.strip() for s in subscriptions_param.split(",") if s.strip()]
        registry = getattr(getattr(request, "app", None), "state", None)
        if registry is not None:
            reg = getattr(registry, "subscription_registry", None)
            if reg is not None:
                return reg.get_all_ids()
        return []
    ```
