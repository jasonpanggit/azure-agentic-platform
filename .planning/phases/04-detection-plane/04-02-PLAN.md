```kql
// classify_domain() — Maps ARM resource_type to AAP domain for agent routing.
// Decision D-05: ARM resource_type is the primary classification signal.
// Decision D-06: Unclassifiable resources fall back to 'sre' domain.
//
// Domains: compute, network, storage, security, arc, sre
// The SRE agent is the catch-all for unrecognized resource types.

.create-or-alter function classify_domain(resource_type: string) {
```

```kql
// EnrichAlerts() — Update policy function: RawAlerts -> EnrichedAlerts
// Extracts resource_name from ARM resource_id. MVP uses alert payload data
// for resource enrichment; full resource inventory join is a Phase 5+ enhancement.

.create-or-alter function EnrichAlerts() {
    RawAlerts
    | extend resource_name = extract(@"/providers/[^/]+/[^/]+/(.+)$", 1, resource_id)
    | extend resource_name = coalesce(resource_name, extract(@"/resourceGroups/([^/]+)", 1, resource_id), "unknown")
    | extend resource_location = ""
    | extend resource_tags = dynamic({})
    | project
        alert_id,
        alert_name,
```

```kql
// ClassifyAlerts() — Update policy function: EnrichedAlerts -> DetectionResults
// Applies classify_domain() to assign each alert to an agent domain.
// All alerts get a domain (D-06: sre is the fallback), so domain IS NOT NULL
// is always true — guaranteeing Activator triggers on every classified row.

.create-or-alter function ClassifyAlerts() {
    EnrichedAlerts
    | extend domain = classify_domain(resource_type)
    | where isnotempty(domain)
    | extend kql_evidence = strcat(
        "Alert: ", alert_name,
        " | Resource: ", resource_id,
        " | Severity: ", severity,
        " | Fired: ", fired_at,
        " | Signal: ", signal_type
    )
    | extend classified_at = now()
    | project
        alert_id,
        severity,
        domain,
        fired_at,
        resource_id,
        resource_type,
        subscription_id,
        resource_name,
        alert_rule,
        description,

}
```

Create `fabric/kql/policies/update_policies.kql`:

```kql
// Update policies — chained pipeline: RawAlerts -> EnrichedAlerts -> DetectionResults
// Chain depth: 2 (max supported: 3)
//
// *** INTENTIONAL: IsTransactional=false on hop 1 (RawAlerts -> EnrichedAlerts) ***
// This OVERRIDES the RESEARCH.md example (which shows IsTransactional: true for both hops).
// Rationale (Risk 6 mitigation from 04-RESEARCH.md):
//   If the EnrichAlerts() function fails and IsTransactional=true, the source ingestion
//   into RawAlerts is also rolled back — causing DATA LOSS of the raw alert.
//   By setting IsTransactional=false, raw data is NEVER lost even if enrichment fails.
//   A failed enrichment will leave the record in RawAlerts only, preservable for
//   manual investigation or re-processing.
//   Validated by test_first_hop_is_non_transactional() in 04-04 test suite.
//
// IsTransactional=true on hop 2 (EnrichedAlerts -> DetectionResults): ensures consistency
// between enriched and classified data. classify_domain() always succeeds (sre fallback),
// so this hop should never fail in practice.

.alter table EnrichedAlerts policy update
@'[{"IsEnabled": true, "Source": "RawAlerts", "Query": "EnrichAlerts()", "IsTransactional": false, "PropagateIngestionProperties": true}]'

.alter table DetectionResults policy update
@'[{"IsEnabled": true, "Source": "EnrichedAlerts", "Query": "ClassifyAlerts()", "IsTransactional": true, "PropagateIngestionProperties": true}]'
```

```python
"""Python mirror of the KQL classify_domain() function (DETECT-002).

This module provides the same domain classification logic as the KQL
function in fabric/kql/functions/classify_domain.kql. It MUST produce
identical results for the same inputs. Used by:
1. Unit tests to validate classification logic
2. The Fabric User Data Function as a fallback/validation layer
"""
from __future__ import annotations

# Domain classification mapping: ARM resource_type prefix -> AAP domain
# Decision D-05: resource_type is the primary classification signal
# Decision D-06: unrecognized types fall back to "sre"
DOMAIN_MAPPINGS: dict[str, str] = {
    # Compute domain
    "microsoft.compute/virtualmachines": "compute",
    "microsoft.compute/virtualmachinescalesets": "compute",
    "microsoft.compute/disks": "compute",
    "microsoft.batch/batchaccounts": "compute",
    "microsoft.compute/availabilitysets": "compute",
    "microsoft.compute/images": "compute",
    # Network domain
    "microsoft.network/virtualnetworks": "network",
    "microsoft.network/networksecuritygroups": "network",
    "microsoft.network/loadbalancers": "network",
    "microsoft.network/applicationgateways": "network",
    "microsoft.network/azurefirewalls": "network",
    "microsoft.network/publicipaddresses": "network",
    "microsoft.network/trafficmanagerprofiles": "network",
    "microsoft.network/frontdoors": "network",
    "microsoft.network/dnszones": "network",
    "microsoft.network/expressroutecircuits": "network",
    "microsoft.network/vpngateways": "network",
    # Storage domain
    "microsoft.storage/storageaccounts": "storage",
    "microsoft.storage/fileservices": "storage",
    "microsoft.storage/blobservices": "storage",
    "microsoft.storagesync/storagesyncservices": "storage",
    # Security domain
    "microsoft.keyvault/vaults": "security",
    "microsoft.security": "security",
    "microsoft.sentinel": "security",
    # Arc domain (hybrid resources)
    "microsoft.hybridcompute/machines": "arc",
    "microsoft.kubernetes/connectedclusters": "arc",
    "microsoft.azurearcdata": "arc",
}

# SRE fallback domain (D-06)
FALLBACK_DOMAIN = "sre"

# Valid domain values (matches IncidentPayload.domain regex: ^(compute|network|storage|security|arc|sre)$)
VALID_DOMAINS = frozenset({"compute", "network", "storage", "security", "arc", "sre"})


def classify_domain(resource_type: str) -> str:
    """Classify an ARM resource_type into an AAP agent domain.

    Args:
        resource_type: ARM resource type string
            (e.g., "Microsoft.Compute/virtualMachines").

    Returns:
        One of: "compute", "network", "storage", "security", "arc", "sre".
        Always returns a non-empty string (sre fallback for unrecognized types).
    """
    if not resource_type:
        return FALLBACK_DOMAIN

    normalized = resource_type.lower().strip()

    # Exact match first
    if normalized in DOMAIN_MAPPINGS:
        return DOMAIN_MAPPINGS[normalized]

    # Prefix match for broad categories (e.g., "microsoft.security/alerts")
    for prefix, domain in DOMAIN_MAPPINGS.items():
        if normalized.startswith(prefix):
            return domain

    return FALLBACK_DOMAIN
```
